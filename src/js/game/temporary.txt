 
    drawBackground(ctx) {
        for(let i = 0; i < this.trees.length; i++) {
            this.trees[i].draw(ctx);
        }
    }
    
    initTreePositions(numberRows, horizonHeight) {
        this.treePositionsY = [];
        let firstPosition = 700;
        let currentPosition = firstPosition;
        let distance = (firstPosition - horizonHeight)/ numberRows;
        for(let i = 0; i < numberRows; i++) {
            this.treePositionsY.push([currentPosition]);
            this.currentPosition -= distance;
        }
    }

    initTrees() {
        for(let i = 0; i < this.treePositionsY.length; i++) {
            this.trees.push(new TreeRow(this.treePositionsY[i], this.treeSpeed));
        }
    }
